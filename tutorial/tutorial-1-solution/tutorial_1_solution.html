import random

guess = 0
r = random.randint(1,10)
attempts = 1

while guess != r:
    guess = input("Please guess!")
    guess = int(guess)
    if guess > r:
        print ("Smaller!")
        attempts+=1
    if guess < r:
        print("Bigger!")
        attempts+=1
print (f"Great, you found the number ({r}) in {attempts} tries!")
